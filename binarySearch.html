        def binarySearch(list, found):
          start = 0
          end = len(list) - 1

          while start <= end:
            mid = (start + end) // 2
            if found == list[mid]:
              return mid
            elif found > list[mid]:
              start = mid + 1
            else:
            end = mid - 1

        data = [1, 4, 9, 16, 25, 36, 49, 65, 81]
        wantedNum = int(input('찾고 싶은 수 입력: '))
        returnNum = binarySearch(data, wantedNum)

        if returnNum == -1:
          print("찾고자 하는 수 없음")
        else:
          print("%d는 %d번째에 위치" %(wantedNum, returnNum+1))
